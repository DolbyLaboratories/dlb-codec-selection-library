69 private void openFile() {
70
71
72 Intent intent = new Intent(Intent.ACTION_OPEN_DOCUMENT);
73 intent.addCategory(Intent.CATEGORY_OPENABLE);
74 intent.setType("video/mp4");
75
76 // Optionally, specify a URI for the file that should appear in the
77 // system file picker when it loads.
78
79 final Activity context = this;
80
81 ActivityResultLauncher<Intent> mStartForResult = registerForActivityResult(new ActivityResultContracts.StartActivityForResult(),
82 new ActivityResultCallback<ActivityResult>() {
83 @Override
84 public void onActivityResult(ActivityResult result) {
85 if (result.getResultCode() == Activity.RESULT_OK) {
86 Intent inputUri = result.getData();
87
88 try {
89
90 if (inputUri == null) {
91 throw new NullPointerException();
92 }
93
94 CodecBuilderImpl builder = new CodecBuilderImpl(context, "placeholder", inputUri.getData());
95
96 Log.e("Main", "onActivityResult: " + builder.printDolbyCodecSupport());
97
98 String codecName = builder.getCodecName();
99
100 Log.e("Main", "onActivityResult: " + codecName);
101
102 boolean dolby = builder.isDolbySupported();
103
104 text.setText(builder.getCodecName() + "\n" + "Dolby: " + dolby);
105
106 ExampleDecoder decoder = new ExampleDecoder();
107
108 decoder.createCodec(MediaCodec.createByCodecName(codecName));
109
110 //Extractor can now be used to play the video.
111 MediaExtractor extractor = builder.configure(decoder.getCodec(), null, decoder, 1);
112
113 } catch (IOException | CodecBuilderImpl.NoCodecException e) {
114 text.setText("No Codec Found");
115 }
116 }
117 }
118 });
119
120 mStartForResult.launch(intent);
121 }
